41void Weiche::weicheWechsel()
42{
43 boolean _wtstatus = digitalRead(_wt);
44 boolean _wgtstatus = digitalRead(_wgt);
45
46 if (_wtstatus == LOW && _wgtstatus == LOW) // wenn die wt und wgt gedrückt werden
47 {
48 if (_weichenposition == true) // wenn die Weiche auf gerade steht schalte auf gerade
49 {
50 Weiche::weicheKurve(); // stelle die Weiche auf Kurve
51 }
52 if (_weichenposition == false) // wenn die Weiche auf Kurve steht schalte auf Kurve
53 {
54 Weiche::weicheGerade(); // stelle die Weiche auf Gerade
55 }
56 }

59void Weiche::weicheGerade() // die Weiche wird in gerade Lage vesetzt
60{
61
62 if (_weichenposition == false && weichenstatus == 0 && _weichenfestlegung == false && !_flankenschutzWeiche) // wenn die Weiche nicht schaltet und in der Kurve steht

64 _wStartzeit = millis();
65 Actor::digitalSchalten(_weichenPinKurve, HIGH); // Relais werden geschaltet

132void Weiche::weicheSchalten() // lässt die Relais wieder in die unaktiv position zurückfallen
133{
134 unsigned long currentmillis = millis();
135 if (!Weiche::weichenstatus == 0)
136 {
137 if (_weichenausleuchtung == true)
138 Weiche::weichenBlinken();
139 if (currentmillis - _wStartzeit >= _weichentimeout)
140 {
141 if (weichenstatus == 1)
142 _weichenposition = true; // die Weiche steht auf Kurvenlage
143 else if (weichenstatus == 2)
144 _weichenposition = false; // die Weiche liegt gerade
145 weichenstatus = 0;
146 Weiche::weicheRelaisHIGH();
147
148 Serial.print("Weiche ");
149 Serial.print(_wnr);
150 Serial.print(" "); // Ausgabe der Weichenstellung an den Seriellen Monitor
151 if (_weichenposition == true)
152 Serial.println("gerade");
153 if (_weichenposition == false)
154 Serial.println("kurve");
155 }
156 }
157 else
158 {
159 if (_weichenposition == true)
160 Weiche::setWeichenLeds(true);
161 if (_weichenposition == false)
162 Weiche::setWeichenLeds(false);
163 }
164}
