38void schieberegister::setPin(int pinNr, boolean pinStatus)
39{
40 digitalWrite(_st, LOW);
41 _daten[pinNr - 1] = pinStatus;
42
43 for (int index = _anzahlPins - 1; index >= 0; index --) // Die Reihenfolge der leuchtenden LEDs
44 { // durch die for- Schleife einfügen
45
46 digitalWrite(_ds, 0); // DS Pin SH_CP Pin auf "0" (LOW) stellen
47 digitalWrite(_sh, 0); //
48 digitalWrite(_ds, _daten[index]); // In DS Pin schreiben, ob LED an oder aus (0 für LOW, also aus, und 1 für HIGH, also an)
49 digitalWrite(_sh, 1); // SH_CP Pin auf 1 setzen (HIGH) damit die Information am DS Pin gespeichert wird
50 }
51 digitalWrite(_st, HIGH); // Wenn for- Schleife beendet wurde, ST_CP Pin auf 1 (HIGH) setzen damit die Datenübertragen werden
52}
