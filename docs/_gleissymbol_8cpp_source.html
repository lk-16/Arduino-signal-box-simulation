55boolean Gleissymbol::isFree(boolean ignoreFlankenschutz)
56{
57 boolean free = true;
58 if (getFahrstrassenelement())
59 {
60 free = false;
61 }
62 else if (_besetztmelder != nullptr && (_besetztmelder->getFahrstrassenelement() || _besetztmelder->besetztmelderAuslesen(LOW, _weiche))) // wenn besetztmelder kein Fahrstraßenelemnt und nicht besetzt
63 {
64 free = false;
65 }
66 else if (_signal != nullptr && _signal->getSignalsperre()) // das signal nicht gesperrt, wenn vorhanden
67 {
68 free = false;
69 }
70 else if (_weiche != nullptr && ignoreFlankenschutz && !_weiche->getFlankenschutz() && _weiche->getWeichenfestlegung()) // die Weiche nicht festgelegt
71 {
72 free = false;
73 }
74 return free;
75}
